{\LARGE \noindent \underline{\textbf{Question 2}}}\\[1mm]
\textbf{[step 1]}\\
Define the set $\mathcal{G}$ of formulas that uoc \{$\neg$, $\wedge$\}\\
\begin{proofindent}
    Let $\mathcal{G}$ be the smallest set such that\\
    \textsc{Basis:} If $x$ is a propositional variable, then $x \in \mathcal{G}$\\
    \textsc{Induction Step:} If $F_1, F_2 \in \mathcal{G}, \text{ then } \neg F, (F_1 \wedge F_2) \in G$
\end{proofindent}
\\\\\\\\\\
\textbf{[step 2]}\\
WTS: For all $F \in \mathcal{G}$ there exists $F'$ such that \\
$F'$ uoc from \{\underline{1},\ $\ll$\}
\\\\
\textsc{Basis:} Let $F = x$, where $x$ is a propositional variable.\\
\begin{proofindent}
    Let $F' = x$\\
    Then $F'$ uoc \{\underline{1},\ $\ll$\}\\
    and $F' \text{\textsc{ leqv }} F$, as wanted 
\end{proofindent}
\\\\
\textsc{Induction Step:} Let $F_1, F_2 \in \mathcal{G}$\\
\begin{proofindent}
    Suppose $F_1', F_2'$ uoc \{\underline{1},\ $\ll$\} and\\
    $F_1' \text{\textsc{ leqv }} F_1 \text{ and } F_2' \text{\textsc{ leqv }} F_2$ \hspace*{15mm} \textbf{[IH]}
    \\\\
    Case 1: $F = \neg F_1$\\
    \begin{proofindent}
        Let $F' = (\underline{1} F_1 \ll \underline{1} F_1) \ll (F_1\ll \underline{1}F_1)$
        \\\\
        then $F'$ uoc \{\underline{1},\ $\ll$\} and
        \\\\
        {\setstretch{1.0}$\begin{array}{r@{}>{\displaystyle}ll}
            F'& {}=  (\underline{1} F_1' \ll \underline{1} F_1') \ll (F_1'\ll \underline{1}F_1')& [\text{by given}]\\
            & {}\text{\textsc{ leqv }} (\underline{1} F_1 \ll \underline{1} F_1) \ll (F_1\ll \underline{1}F_1) \hspace*{10mm} & [\text{by \textbf{IH}}]\\
            & {}\text{\textsc{ leqv }}  (\underline{1} F_1 \ll \underline{1} F_1) \ll (\neg F_1) & [\text{since $F_1 \ll \underline{1}F_1$ sastified if $F_1$ falsified}]\\
            & {}\text{\textsc{ leqv }}  (\underline{0} F_1 ) \ll (\neg F_1) & [\text{since $x \ll x$ always falsified}]\\
            & {}\text{\textsc{ leqv }}  \neg F_1 & [\text{since $\underline{0} F_1  \ll \neg F_1$ sastified if $\neg F_1$ sastified}]\\
            & {}= F & [\text{by given}]\\
        \end{array}$}\\
        as wanted.
    \end{proofindent}
    \\\\\\
    Case 2: $F =  F_1 \wedge F_2$\\
    \begin{proofindent}
        Let $F' = (F_1 \ll \underline{1}F_1) \ll F_2$
        \\\\
        Then $F'$ uoc \{\underline{1},\ $\ll$\} and 
        \\\\
        {\setstretch{1.0}$\begin{array}{r@{}>{\displaystyle}ll}
            F'& {}=  (F_1' \ll \underline{1}F_1') \ll F_2' & [\text{by given}]\\
            & {}\text{\textsc{ leqv }}  (F_1 \ll \underline{1}F_1) \ll F_2 \hspace*{10mm} & [\text{by \textbf{IH}}]\\
            & {}\text{\textsc{ leqv }}  (\neg F_1) \ll F_2 & [\text{since $F_1 \ll \underline{1}F_1$ sastified if $F_1$ falsified}]\\
            & {}\text{\textsc{ leqv }}  F_1 \wedge F_2 & [\text{since they have the same truth tables}]\\
            & {}= F &\text{[by given]}\\

        \end{array}$}\\
        as wanted.
    \end{proofindent}
\end{proofindent}
\\\\\\\\\\
\textbf{[step 3]}\\
Since \{$\neg$, $\wedge$\} is complete, therefore \{\underline{1}, $\ll$\} is also complete. \qed
